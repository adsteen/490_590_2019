library(tidyverse)

# Let's talk about linear models

glimpse(mtcars)

# What is the relationship between power and displacement?
p_hp <- ggplot(mtcars, aes(x = disp, y = hp)) +
  geom_point() + 
  geom_smooth(method = "lm", se = TRUE) + 
  ggrepel::geom_label_repel(label = rownames(mtcars))
print(p_hp)

# But what's the relationship?
hp_mod <- lm(hp ~ disp, data = mtcars) # lm for "linear model"
str(hp_mod)
summary(hp_mod)

# Note that summary is doing something funny here:
summary(mtcars$hp)

# an even better example
print(mtcars$mpg)
print(p_hp)

library(datasauRus)
ggplot(datasaurus_dozen, aes(x = x, y = y)) + 
  geom_point() + 
  geom_smooth(method = "lm", se = TRUE) + 
  facet_wrap(~dataset)

x_data <- datasaurus_dozen %>%
  filter(dataset == "x_shape")
x_mod <- lm(y ~ x, data = x_data)
summary(x_mod)
plot(x_mod)
